第一题：
```js
function A() {
  for(var i = 0; i < 5; i ++) {
    setTimeout(function(){
      console.log(i)
      console.log(new Date().valueOf())
    }, 2000)
  }
}
A()
console.log(i)
console.log(new Date().valueOf())

//10行的i是未定义，所以11行没执行
//9行 输出为：5，2s，循环5次
```

}

  console.log('bbb')
}
printB();

  console.log('ddd')
}
printD();

printC();
var printC = () => {
  console.log('ccc')
}

第三题：
组件关系如下，a-b-(c,d)，说明挂载生命周期
```js
A beforeCreate
A created
A beforeMount
B beforeCreate
B created
B beforeMount
D beforeCreate
D created
D beforeMount
C beforeCreate
C created
C beforeMount
D mounted
C mounted
B mounted
A mounted
```
C和D的顺序按照在template中出现的顺序来定
