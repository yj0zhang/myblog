第一题：
```js
function A() {
  for(var i = 0; i < 5; i ++) {
    setTimeout(function(){
      console.log(i)
      console.log(new Date().valueOf())
    }, 2000)
  }
}
A()
console.log(i)
console.log(new Date().valueOf())

//10行的i是未定义，所以11行没执行
//9行 输出为：5，2s，循环5次
```
